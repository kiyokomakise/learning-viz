URLは変わらないので再設定不要 // ════════════════════════════════════════════════════════════════ const SHEET_SKILLS = '理解度データ'; const SHEET_LOG = '学習ログ'; const SHEET_SUMMARY = '集計サマリー'; // ── 受信（受講生からのデータ保存）────────────────────────────── function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); // learning / learning2 / learning3 すべて対応 if (data.type === 'learning' || data.type === 'learning2' || data.type === 'learning3') { saveLearningData(ss, data); } return ContentService .createTextOutput(JSON.stringify({ status: 'ok' })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', message: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ── 送信（受講生・管理者からのデータ取得）───────────────────── function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const action = e.parameter.action || 'getStudent'; let result = {}; if (action === 'getStudent') { result = getStudentData(ss, e.parameter.studentId); } else if (action === 'getAll' || action === 'summary') { result = getAllStudents(ss); } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', message: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ── 学習データを保存 ───────────────────────────────────────── function saveLearningData(ss, data) { // ① 理解度データシート（最新状態を受講生ごとに1行で保持） let skillSheet = ss.getSheetByName(SHEET_SKILLS); if (!skillSheet) { skillSheet = ss.insertSheet(SHEET_SKILLS); skillSheet.appendRow([ 'タイムスタンプ', '受講生番号', '氏名', '総合（全体）', 'Word習得数', 'Excel習得数', 'PP習得数', 'Illustrator習得数', 'Photoshop習得数', 'Web習得数', 'ビジマナ習得数', 'スキルデータ(JSON)', '学習ログ(JSON)' ]); skillSheet.setFrozenRows(1); skillSheet.getRange(1,1,1,13).setBackground('#2E4057').setFontColor('#ffffff').setFontWeight('bold'); } // スキルデータ取得（skillData形式: { id: { lv, date } }） const skillData = data.skillData || data.skills || {}; // 科目別lv3カウント（各科目のID prefix で判定） const prefixes = { word:'w', excel:'e', pp:'p', illust:'i', photo:'ph', web:'wb', manner:'m' }; const counts = {}; Object.entries(prefixes).forEach(([subj, prefix]) => { counts[subj] = Object.entries(skillData).filter(([id, v]) => id.startsWith(prefix) && v.lv === 3 ).length; }); // 総合lv3数 const totalLv3 = Object.values(skillData).filter(v => v.lv === 3).length; // 既存行を検索（01と1など表記ゆれも吸収） const vals = skillSheet.getDataRange().getValues(); let targetRow = -1; const normalId = String(parseInt(data.studentId)); // "01"→"1" に正規化 for (let i = 1; i < vals.length; i++) { const rowId = String(parseInt(String(vals[i][1]))); // セルの値も正規化 if (rowId === normalId) { targetRow = i + 1; break; } } const row = [ new Date(), data.studentId, data.studentName || '', totalLv3, counts.word || 0, counts.excel || 0, counts.pp || 0, counts.illust || 0, counts.photo || 0, counts.web || 0, counts.manner || 0, JSON.stringify(skillData), JSON.stringify(data.history || data.studyLog || {}), ]; if (targetRow > 0) { skillSheet.getRange(targetRow, 1, 1, row.length).setValues([row]); } else { skillSheet.appendRow(row); } // ② 学習ログシート（保存履歴を追記） let logSheet = ss.getSheetByName(SHEET_LOG); if (!logSheet) { logSheet = ss.insertSheet(SHEET_LOG); logSheet.appendRow(['タイムスタンプ', '受講生番号', '氏名', '総習得数', 'Word', 'Excel', 'PP', 'Illustrator', 'Photoshop', 'Web', 'ビジマナ']); logSheet.setFrozenRows(1); logSheet.getRange(1,1,1,11).setBackground('#2E4057').setFontColor('#ffffff').setFontWeight('bold'); } logSheet.appendRow([ new Date(), data.studentId, data.studentName || '', totalLv3, counts.word, counts.excel, counts.pp, counts.illust, counts.photo, counts.web, counts.manner ]); // ③ 集計サマリー更新 updateSummary(ss); } // ── 受講生個人データを返す ───────────────────────────────────── function getStudentData(ss, studentId) { if (!studentId) return { status: 'error', message: 'studentId required' }; const sheet = ss.getSheetByName(SHEET_SKILLS); if (!sheet || sheet.getLastRow() <= 1) return { status: 'ok', skillData: {}, history: [] }; const vals = sheet.getDataRange().getValues(); const normalStudentId = String(parseInt(studentId)); for (let i = 1; i < vals.length; i++) { const rowId = String(parseInt(String(vals[i][1]))); if (rowId === normalStudentId) { let skillData = {}; let history = []; try { skillData = JSON.parse(vals[i][11] || '{}'); } catch(e){} try { history = JSON.parse(vals[i][12] || '[]'); } catch(e){} // historyがオブジェクト形式の場合は配列に変換 if (!Array.isArray(history) && typeof history === 'object') { history = Object.entries(history).map(([date,avg])=>({ date, totalLv3: avg })); } return { status: 'ok', studentId: vals[i][1], studentName: vals[i][2], skillData, history, }; } } return { status: 'ok', skillData: {}, history: [] }; } // ── 全受講生データを返す（管理者用）────────────────────────── function getAllStudents(ss) { const sheet = ss.getSheetByName(SHEET_SKILLS); if (!sheet || sheet.getLastRow() <= 1) return { status: 'ok', students: [] }; const vals = sheet.getDataRange().getValues(); const students = []; for (let i = 1; i < vals.length; i++) { students.push({ studentId: String(vals[i][1]), studentName: vals[i][2], totalLv3: vals[i][3], word: vals[i][4], excel: vals[i][5], pp: vals[i][6], illust: vals[i][7], photo: vals[i][8], web: vals[i][9], manner: vals[i][10], updatedAt: vals[i][0] ? new Date(vals[i][0]).toLocaleDateString('ja-JP') : '', }); } return { status: 'ok', students: students.sort((a,b)=>Number(a.studentId)-Number(b.studentId)), updatedAt: new Date().toLocaleString('ja-JP'), }; } // ── 集計サマリー自動更新 ────────────────────────────────────── function updateSummary(ss) { let sheet = ss.getSheetByName(SHEET_SUMMARY); if (!sheet) sheet = ss.insertSheet(SHEET_SUMMARY); sheet.clearContents(); const skillSheet = ss.getSheetByName(SHEET_SKILLS); if (!skillSheet || skillSheet.getLastRow() <= 1) return; const vals = skillSheet.getDataRange().getValues(); const data = vals.slice(1); if (!data.length) return; const n = data.length; const avg = (col) => Math.round(data.reduce((s,r)=>s+Number(r[col]||0),0)/n); const avgTotal = avg(3); const avgWord = avg(4); const avgExcel = avg(5); const avgPP = avg(6); const avgIllust= avg(7); const avgPhoto = avg(8); const avgWeb = avg(9); const avgManner= avg(10); sheet.getRange('A1').setValue('集計サマリー（自動更新）').setFontSize(14).setFontWeight('bold').setFontColor('#2E4057'); sheet.getRange('A2').setValue('最終更新：' + new Date().toLocaleString('ja-JP')).setFontColor('#888888'); const summaryData = [ ['', ''], ['受講生数', n + '名'], ['総習得数（平均）', avgTotal + '項目'], ['Word 習得数（平均）', avgWord + '項目'], ['Excel 習得数（平均）', avgExcel + '項目'], ['PowerPoint 習得数（平均）', avgPP + '項目'], ['Illustrator 習得数（平均）', avgIllust + '項目'], ['Photoshop 習得数（平均）', avgPhoto + '項目'], ['Web 習得数（平均）', avgWeb + '項目'], ['ビジマナ 習得数（平均）', avgManner + '項目'], ]; sheet.getRange(3, 1, summaryData.length, 2).setValues(summaryData).setFontSize(12); sheet.setColumnWidth(1, 240); sheet.setColumnWidth(2, 120); }
